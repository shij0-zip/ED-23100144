public class RecursivoCodigo 
{

    public static long fac_iterativo(int n) {
        long Res = 1;

        for (int i = 2; i <= n; i++) {
            Res = Res * i;
        }
        
        return Res;
    }

    public static long fac_recursivo(int x) {
        if (x == 1 || x == 0) {
            return 1; // Return x 
        }

        return x * fac_recursivo(x - 1);
    }
}
