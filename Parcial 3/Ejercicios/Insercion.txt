public class InsertionSortCodigo 
{

    public static int[] ins(int[] arr) {
        for (int k = 1; k < arr.length; k++) {
            int Key = arr[k];
            int j = k - 1;

            while (j >= 0 && arr[j] > Key) {
                arr[j + 1] = arr[j];
                j = j - 1;
            }
            
            arr[j + 1] = Key;
        }
        
        return arr;
    }
}
